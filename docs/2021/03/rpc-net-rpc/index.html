	fmt.Println("server recv:", req)
	*reply = 2
	return nil
}

func server() {
	ls, err := net.Listen("tcp", ":9090")
	if err != nil {
		panic(err)
	}
	rpc.Register(new(ServiceA))
	for {
		rpc.Accept(ls)
	}
}

func main() {
	go server()

		panic(err)
	}

		panic(err)
	}
	fmt.Println("client: recv:", ax)
}

func (c *gobClientCodec) WriteRequest(r *Request, body interface{}) (err error) {
