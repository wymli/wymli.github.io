package main

import (
	"fmt"
	"math/rand"
	"reflect"
	"sort"
)

func swap(a []int, i, j int) {
	a[i], a[j] = a[j], a[i]
}

func partition(a []int) int {
	r := rand.Int() % len(a)
	swap(a, r, len(a)-1)
	// jæŒ‡é’ˆéå†æ•°ç»„a
	// iæŒ‡é’ˆ a[:i]å‡æ˜¯æ¯”a[len(a)-1]å°çš„æ•°
	i, j := 0, 0
	for j < len(a)-1 {
		if a[j] < a[len(a)-1] {
			swap(a, i, j)
			i++
		}
		j++
	}
	swap(a, i, len(a)-1)
	return i
}

func qsort(a []int) {
	if len(a) == 0 {
		return
	}
	mid := partition(a)
	qsort(a[:mid])
	qsort(a[mid+1:])
}

func main() {
	a := []int{rand.Int(), rand.Int(), rand.Int(), rand.Int(), rand.Int(), rand.Int(), rand.Int(), rand.Int()}
	b := make([]int, len(a))
	copy(b, a)
	sort.Ints(a)
	qsort(b)
	fmt.Println(reflect.DeepEqual(a, b))
}

func merge(a, b []int) {

func msort(a []int) {
	if len(a) == 0 || len(a) == 1{
		return 
	}
	left := a[:len(a)/2]
	right := a[len(a)/2:]
	msort(left)
	msort(right)
	merge(left, right)
}
